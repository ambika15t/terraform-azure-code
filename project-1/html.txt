const express = require('express');
const os = require('os');
const app = express();
const port = 3000;

app.get('/', (req, res) => {
    const hostname = os.hostname();
    res.send(`
        <!DOCTYPE html>
        <html>
        <head>
            <title>Hello World</title>
        </head>
        <body>
            <h1>Hello World</h1>
            <p>Hostname: ${hostname}</p>
        </body>
        </html>
    `);
});

app.listen(port, () => {
    console.log(\`Server running at http://localhost:\${port}\`);
});
